\section*{Misiones en números}

\begin{frame}{El Bosque Atlántico}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{El marco ecológico global}
      \begin{itemize}
        \item Originalmente: \num{1.3} millones de \unit{\km\squared} \footcite{ribeiro2009atlantic}
        \item Brasil (\qty{92}{\percent}), Paraguay (\qty{6}{\percent}), Argentina (\qty{2}{\percent})
        \item Hoy: solo \qtyrange{12}{17}{\percent} de su extensión original
        \item Uno de los \textbf{focos de biodiversidad} más amenazados del planeta \footcite{wwf2024atlantic}
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      \includegraphics[width=0.95\textwidth]{img/bosque_altantico.png}
      
      \vspace{0.2em}
      {\tiny Fuente: \href{https://www.researchgate.net/figure/Figura-8-Proceso-de-deforestacion-del-Bosque-Atlantico-Alto-Parana-desde-el-ano-1900_fig1_268334207}{ResearchGate (Fig. 8)}}
    \end{column}
  \end{columns}
  \note{
    Arrancamos con el contexto global. El Bosque Atlántico es uno de los focos de biodiversidad más amenazados del planeta, habiéndose perdido más del 80\% de su cobertura original. Si miran la imagen del proceso de deforestación, el pequeño remanente continuo que sobrevive en la frontera de Argentina es Misiones. Por eso, en la siguiente diapositiva, hacemos foco en esta selva.
  }
\end{frame}

\begin{frame}{Importancia Ecológica}
  \textbf{Servicios ecosistémicos:}
  \begin{itemize}
    \item Regulación del ciclo hidrológico
    \item Secuestro de carbono
    \item Protección contra la erosión del suelo
    \item Regulación climática regional
  \end{itemize}

  \vspace{1em}
  \textbf{Marco legal:} Ley 26.331 de Protección de Bosques Nativos \footcite{ley26331bosques}
  \begin{description}
    \item[Categoría I (Rojo):] Muy alto valor - no se transforma
    \item[Categoría II (Amarillo):] Mediano valor - uso sostenible
    \item[Categoría III (Verde):] Bajo valor - puede transformarse
  \end{description}
  \note{
    Además del valor biológico, el bosque provee servicios ecosistémicos
    críticos. La Ley 26.331 establece categorías de protección. En Misiones,
    la mayoría del territorio está en categorías I y II, lo que limita la
    transformación del bosque.
  }
\end{frame}

\section*{Amenazas}
\begin{frame}{Cambio de Uso del Suelo y Degradación Forestal}
  \begin{itemize}
    \item \textbf{Expansión agrícola:} Grandes monocultivos orientados a la exportación (soja, caña de azúcar, café).
    \item \textbf{Pastoreo de ganado:} Conversión masiva de la selva nativa en pastizales.
    \item \textbf{Fragmentación del hábitat:} Reducción del bosque a ``islas'' aisladas, limitando el movimiento de las especies y su viabilidad genética.
    \item \textbf{Efecto de borde:} El aumento de la relación perímetro/superficie altera el microclima, reduce la humedad y facilita la entrada de especies invasoras.
  \end{itemize}
  \note{
    Explicar que la conversión de bosques para agricultura comercial y ganadería es la causa principal de la pérdida y degradación. Genera un efecto isla que fragmenta los ecosistemas.
  }
\end{frame}

\begin{frame}{Deforestación y Extracción Ilegal de Madera}
  \begin{columns}
    \begin{column}{0.4\textwidth}
      \begin{figure}
        \centering
        \includegraphics[width=\textwidth]{img/deforestacion.png}
      \end{figure}
    \end{column}
    \begin{column}{0.6\textwidth}
      \begin{itemize}
        \item Incursiones constantes en áreas protegidas para la tala ilegal.
        \item Degradación de la estructura interna del bosque y pérdida de biomasa.
        \item Apertura de picadas y caminos que facilitan el acceso a zonas antes inaccesibles, promoviendo nuevas intrusiones.
        \item Entre 1990 y 2020 se perdieron $\sim$\qty{130000}{ha} de Corredor Verde \footcite{fauba2024corredorverde}.
      \end{itemize}
    \end{column}
  \end{columns}
  \note{
    Mencionar que la tala ilegal daña gravemente la estructura interna del bosque. Incorporar las cifras locales de pérdida en el Corredor Verde de Misiones (130 mil ha en 30 años).
  }
\end{frame}

\begin{frame}{Expansión Urbana e Industrialización}
  \begin{columns}
    \begin{column}{0.55\textwidth}
      \begin{itemize}
        \item \textbf{Contaminación:} Vertidos de aguas residuales sin tratar y residuos industriales en cuencas hidrográficas.
        \item \textbf{Infraestructura fragmentadora:} Construcción de viviendas y carreteras que cortan los corredores biológicos.
        \item \textbf{Atropellamientos de fauna:} Más de \num{5000} animales al año mueren en rutas que cruzan áreas protegidas \footcite{vidasilvestre2024atropellamiento}.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{figure}
        \centering
        \includegraphics[width=\textwidth]{img/atropello.png}
      \end{figure}
    \end{column}
  \end{columns}
  \note{
    La expansión urbana e industrial presiona mediante la contaminación del agua y la fragmentación física por carreteras, aislando poblaciones y aumentando atropellamientos.
  }
\end{frame}

\begin{frame}{Actividades Ilícitas y Extracción de Recursos}
  \begin{itemize}
    \item \textbf{Caza furtiva y trampeo:} Incursiones armadas para la caza deportiva o comercial, y captura de animales exóticos para el tráfico de fauna.
    \item \textbf{Degradación selectiva de flora:} Extracción ilegal y hormiga de especies nativas de alto valor comercial (plantas ornamentales, medicinales y subproductos forestales).
    \item \textbf{Contrabando:} Aprovechamiento de la porosidad de los límites fluviales con Brasil y Paraguay para la evasión de controles fronterizos.
  \end{itemize}
  \note{
    La caza furtiva, plantaciones clandestinas, robo de palmito y contrabando fronterizo explotan áreas remotas. El patrullaje tradicional es ineficiente por la falta de detección en tiempo real, lo que justifica la propuesta de este proyecto.
  }
\end{frame}

\begin{frame}{Uso de Cámaras Trampa}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{Funcionamiento:}
      \begin{itemize}
        \item Activación por sensor \textbf{PIR}
        \item Captura al detectar presencia
      \end{itemize}

      \vspace{0.5em}
      \textbf{Modalidades:}
      \begin{itemize}
        \item \textbf{Pasivas:} Almacenamiento local (tarjeta SD)
        \item \textbf{Activas:} Transmisión (celular o satelital)
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      \begin{tikzpicture}[
          font=\sffamily,
          >=Latex,
          camera/.style={draw=red!80!black, fill=red!15, rounded corners=4pt, minimum width=0.9cm, minimum height=1.6cm, line width=1.5pt},
          detector/.style={fill=blue!20!gray, draw=blue!40!gray, opacity=0.35, line width=1pt, dashed}
        ]
        % Área de detección
        \fill[detector] (0.45,0) -- +(40:3.5cm) arc (40:-40:3.5cm) -- cycle;
        \node[text=blue!60!black, font=\small\bfseries, align=center] at (2.2,0) {Área de\\detección};
        
        % Objeto en movimiento (entrando al área)
        \draw[gray!60, line width=1.1pt, dashed, ->] (3.8,2.3) -- (3.1,1.8);
        \node[circle, fill=red!80, minimum size=0.15cm, inner sep=0pt] (obj) at (3.1,1.8) {};
        \draw[red!80!black, thick] (obj) circle (4pt);

        % Cuerpo de la cámara
        \node[camera] (cam) at (0,0) {};
        
        % Lente y sensor PIR (detalles de la cámara)
        \fill[red!80!black] (0.45,0.25) rectangle (0.55,-0.25);
        \fill[black!80] (0.45,0.1) circle (2.2pt);
        \fill[white!90!gray] (0.45,-0.17) rectangle (0.53,-0.07);
        
        % Etiqueta de la cámara
        \node[below=0.2cm of cam, font=\small\bfseries, text=black!80, align=center] {Cámara\\trampa};
      \end{tikzpicture}
    \end{column}
  \end{columns}
  \note{
    Las cámaras trampa se activan por movimiento mediante sensores PIR y capturan imágenes al detectar presencia (fauna o intrusos).
    Existen dos modalidades:
    - Pasivas: Guardan fotos localmente en tarjeta SD (requieren recolección manual, lo que genera alta latencia).
    - Activas: Transmiten fotos al instante usando redes 3G/4G o satelitales (pero requieren infraestructura costosa o inexistente en selva profunda).
  }
\end{frame}

\begin{frame}{Limitaciones del Monitoreo Actual}
  \begin{block}{Cámaras Pasivas: El problema de la latencia}
    ¿Cuánto tiempo transcurre entre una intrusión y su detección?
  \end{block}

  \begin{block}{Cámaras Activas: Infraestructura y Costos}
    \begin{itemize}
      \item \textbf{Dependencia celular:} Sin señal en selva profunda.
      \item \textbf{Costos elevados:} Suscripciones satelitales y hardware costoso.
      \item \textbf{Escasa escalabilidad:} Inviable para despliegues masivos.
    \end{itemize}
  \end{block}
  \note{
    Las cámaras pasivas tienen el gran problema de la latencia. Ante la pregunta: ¿Cuánto tiempo transcurre entre una intrusión y su detección?, la respuesta es: semanas o meses, hasta recolectar la tarjeta SD de forma manual. Para entonces, el daño ya está hecho.

    Las cámaras activas resuelven la latencia pero fallan en la práctica por:
    1. Falta de señal celular en zonas interiores.
    2. Costo prohibitivo de planes satelitales y de hardware especial.
    3. Imposibilidad de escalar a decenas o cientos de nodos por presupuesto.
  }
\end{frame}

  \end{itemize}

\section*{El Rol de la Tecnología}

\begin{frame}{Avances en Hardware y Sensores de Campo}
  \textbf{Dispositivos de captura autónomos y procesamiento local:}
  
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{Cómputo en el Borde (Edge Computing):} Microcomputadoras y microcontroladores de bajo costo capaces de procesar datos localmente en el nodo sensor.
    \item \textbf{Cámaras Trampa Modernas:} Dispositivos con sensores de movimiento PIR optimizados, flash infrarrojo invisible y alta eficiencia energética para operación continua por meses.
    \item \textbf{Sistemas Autónomos Integrados:} Prototipos académicos (ej. \textit{AiCatcher} \footcite{mallya2019aicatcher}) orientados a la inferencia y captura automatizada.
  \end{itemize}
  
  \note{
    Resumir los avances físicos: microcomputadoras minúsculas de bajo costo y consumo en el borde, mejoras críticas en las cámaras trampa tradicionales para capturar imágenes sin alertar a la fauna, y el desarrollo de prototipos integrados autónomos.
  }
\end{frame}

\begin{frame}{Avances en Conectividad e Internet de las Cosas (IoT)}
  \textbf{Comunicación y transmisión de datos en áreas sin cobertura comercial:}
  
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{Redes Autoorganizadas (Mesh):} Redes descentralizadas basadas en protocolos como WiFi o Bluetooth, donde cada sensor actúa como repetidor para extender la cobertura.
    \item \textbf{Protocolos de Largo Alcance (LoRaWAN):} Tecnologías de modulación de radio de bajo consumo que enlazan sensores a kilómetros de distancia con un ancho de banda mínimo.
    \item \textbf{Telemetría Satelital Híbrida:} Integración de módulos satelitales compactos (ej. red Iridium) para transmitir alertas críticas en tiempo real desde selvas densas o aisladas.
  \end{itemize}
  
  \note{
    Explicar la evolución de las telecomunicaciones aplicadas a la conservación: redes mesh para autoorganizarse en terreno difícil, protocolos de largo alcance como LoRa para enviar datos a gran distancia con baterías chicas, y soluciones satelitales híbridas.
  }
\end{frame}

\begin{frame}{Avances en Software y Visión Computacional}
  \textbf{Automatización y procesamiento masivo de datos ecológicos:}
  
  \vspace{0.5em}
  \begin{itemize}
    \item \textbf{Ecosistemas en la Nube:} Plataformas globales como \textit{Microsoft WildLife} (e iniciativas derivadas de \textit{AI for Earth}) que centralizan datos geoespaciales y facilitan herramientas de computación a investigadores de todo el mundo.
    \item \textbf{Detección en Tiempo Real (YOLO):} Modelos de redes neuronales convolucionales aplicados a video y fotografía que permiten clasificar e identificar múltiples objetos (fauna, vehículos, personas) de forma simultánea e instantánea.
    \item \textbf{Modelos Pre-entrenados (MegaDetector):} Algoritmos abiertos especializados en el filtrado masivo de imágenes de cámaras trampa; automatizan la eliminación de falsos positivos (como movimiento de vegetación o sombras) para aislar la actividad biológica relevante.
  \end{itemize}

  \note{
    Explicar el salto cualitativo en el procesamiento: la IA resuelve el cuello de botella de analizar millones de fotos o videos. Modelos como YOLO permiten la detección inmediata de objetos, mientras que herramientas específicas de conservación como MegaDetector limpian de forma automatizada las ráfagas de fotos vacías causadas por el viento o la luz.
  }
\end{frame}

\begin{itemize}
\item Solucion sobre WiFi $2.4$GHz (Modo SoftAP + Station)
\item Extiende el alcance sin depender de infraestructura celular
\item Comportamiento de router: el reenvio de paquetes de otros nodos ocurre transparentemente a nivel de red (IP)
\end{itemize}

\vspace{1em}
\begin{alertblock}{Flujo de Datos Transparente}
Cuando el PIR detecta movimiento, el nodo despierta y realiza una \textbf{peticion HTTP POST}. A partir de ahi, los nodos intermedios simplemente leen la direccion IP destino y enrutan los paquetes salto por salto hasta el servidor.
\end{alertblock}
\note{
Para sacar la foto de la selva usamos ESP-MESH-LITE. Un detalle clave a 
destacar: los nodos actúan literalmente como routers. La capa de
aplicación de cada nodo solo se encarga de originar su propia foto. Cuando un
nodo ``Relay'' recibe paquetes de otros nodos, los reenvía directamente
viendo la IP, sin involucrar a su capa de aplicación.
}
\end{frame}

\begin{frame}{Red Malla - Roles y Energía}
  \textbf{Trade-off de Energía y Funcionalidad:}
  
  \vspace{1em}
  \begin{columns}[t]
    \begin{column}{0.32\textwidth}
      \begin{block}{Leaf}
        \small
        \textbf{Bajo consumo}
        \begin{itemize}
          \scriptsize
          \item Deep Sleep + ULP
          \item Sin enrutar tráfico
          \item Máxima autonomía
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{block}{Relay}
        \small
        \textbf{Nodo intermedio}
        \begin{itemize}
          \scriptsize
          \item Light Sleep
          \item Enruta tráfico
          \item Antena: Externa omnidireccional
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.32\textwidth}
      \begin{block}{Root}
        \small
        \textbf{Nodo Raíz}
        \begin{itemize}
          \scriptsize
          \item 100\% activo
          \item Puerta al Router
          \item Sin cámara (ESP32)
        \end{itemize}
      \end{block}
    \end{column}
  \end{columns}
  \note{
    Explicamos los tres roles de la red malla y su balance energético.
    1. El Leaf prioriza el ahorro de batería (Deep sleep con ULP), no ayuda a enrutar y tiene la mayor autonomía.
    2. El Relay sacrifica batería permaneciendo en Light sleep para poder recibir y enrutar paquetes de otros nodos.
    3. El Root es la única puerta de enlace, está 100% activo sin cámara y conectado directamente a una fuente de alimentación o router.
  }
\end{frame}

\begin{frame}{Red Malla - Robustez (Self-Healing)}
  \begin{columns}[c]
    \begin{column}{0.5\textwidth}
      \textbf{Auto-recuperación (Self-Healing):}
      \vspace{1em}
      
      \only<1>{
        \textbf{1. Estado Normal}
        \begin{itemize}
          \item Estructura jerárquica activa.
          \item El nodo Leaf C envía datos a través del Relay A.
        \end{itemize}
      }
      \only<2>{
        \textbf{2. Pérdida de Enlace}
        \begin{itemize}
          \item El nodo Relay A falla o se apaga.
          \item C detecta la pérdida del enlace y queda desconectado.
        \end{itemize}
      }
      \only<3>{
        \textbf{3. Re-escaneo}
        \begin{itemize}
          \item C realiza un escaneo activo del entorno de radio.
          \item Busca balizas e identifica al Relay B como nuevo padre potencial.
        \end{itemize}
      }
      \only<4>{
        \textbf{4. Conexión Restablecida}
        \begin{itemize}
          \item C se asocia con el Relay B.
          \item La red se reestructura automáticamente.
        \end{itemize}
      }
    \end{column}

\begin{frame}{Red Malla - Selección de Root Node}
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \textbf{Elección del nodo Raíz (Root):}
      \vspace{0.8em}
      \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \small
        \item Los candidatos evalúan su potencia de señal \textbf{RSSI} al Router.
        \item El nodo con mejor señal (\textbf{-58 dBm}) se establece automáticamente como \textbf{Root}.
        \item Nodos con señal más débil (\textbf{-85 dBm}) se reconfiguran como \textbf{Relays} y se conectan al Root.
      \end{itemize}
      
      \vspace{0.8em}
      \begin{block}{Prevención de Bucles}
        \scriptsize
        Evita reenvíos cíclicos de paquetes durante la autoconfiguración jerárquica de la topología.
      \end{block}

\begin{frame}{Lógica de Umbrales de Detección}
\textbf{Tres niveles de decisión para enviar alertas:}

\vspace{0.5em}
\begin{description}
\item[HIGH = 0{,}80:] Confianza $>$ \qty{80}{\percent} $\rightarrow$ alerta \textbf{inmediata}, sin importar clasificación
\item[MID = 0{,}50:] Confianza entre \qty{50}{\percent} y \qty{80}{\percent} y clasificador dice ``vacío'' $\rightarrow$ alerta igualmente (posible animal pequeño)
\item[LOW = 0{,}30:] Confianza $<$ \qty{30}{\percent} y clasificador ``vacío'' $\rightarrow$ se \textbf{descarta} (probable falso positivo del PIR)
\end{description}

\vspace{0.5em}
\begin{alertblock}{Política}
Sesgo hacia la \textbf{sensibilidad}: ante la duda, se transmite la imagen y se delega la decisión al operador humano.
\end{alertblock}
\note{
Tres umbrales configurables: HIGH (80\%) envía siempre, MID (50\%) envía
si el clasificador dice vacío (posible animal pequeño), LOW (30\%) descarta.
Política conservadora: preferimos falsos positivos a perder eventos reales.
}
\end{frame}

    En condiciones controladas de laboratorio (red LAN local, servidor en la
    misma subred), el sistema logra una latencia de transmisión HTTP de 463~ms
    para una imagen de 12.9~KB, con un ciclo completo wake-up-to-sleep de solo
    1.27~s, lo cual valida la eficiencia del firmware.

    La prueba de campo se realizó intencionalmente en condiciones de
    conectividad adversas: un hotspot de celular con señal débil,
    compartido simultáneamente entre los nodos mesh, la computadora
    ejecutando el servidor HTTP, el bot de Telegram y otros servicios.
    Esta saturación del canal inalámbrico y la alta latencia del uplink
    celular explican la diferencia de dos órdenes de magnitud respecto
    a la prueba LAN. El valor de 40~s promedio refleja las limitaciones
    del enlace de salida (backhaul), no del protocolo mesh ni del firmware.

    En un despliegue real con un router WiFi dedicado (o incluso un
    hotspot con mejor cobertura), se esperaría un rendimiento
    significativamente más cercano al obtenido en laboratorio.

\begin{frame}{Limitaciones Conocidas}
\begin{description}
\item[Operación diurna:] Lente con filtro IR, sin visión nocturna
\item[Sensor PIR:] Optimizado para humanos, puede no detectar fauna pequeña
\item[Resolución:] VGA (640$\times$480), limitada por el balance entre calidad JPEG y tiempo de transmisión
\item[Consumo:] WiFi consume más que cámaras trampa comerciales optimizadas
\item[Gestión de energía:] Nodos intermedios (Relay) deben mantener conectividad mesh permanente, limitando su autonomía
\end{description}

\vspace{0.5em}
\begin{block}{Nota sobre ``tiempo real''}
No es tiempo real deterministico (milisegundos), sino \textbf{operativo}: de dias/semanas a \textbf{minutos}.
\end{block}
\note{
Transparencia sobre limitaciones: sin visión nocturna, PIR domótico,
resolución VGA limitada por el balance compresión/transmisión.
Nodos Relay deben mantener la red activa (Light Sleep), lo que reduce
autonomía. Tiempo real significa minutos, no milisegundos.
}
\end{frame}

\begin{frame}{Trabajos Futuros}

\item[Visión nocturna:] Cámaras con capacidad infrarroja

\item[Optimización energética:] Paneles solares, reguladores más eficientes

\item[Procesamiento en borde:] TinyML en nodos de cámara

\end{description}
